考虑朴素的n2n^2n2方程，dp[i]=min⁡(dp[j]+M+(sum[i]−sum[j])2)dp[i]=\min(dp[j]+M+(sum[i]-sum[j])^2)dp[i]=min(dp[j]+M+(sum[i]−sum[j])2)

其中sum[i]sum[i]sum[i]为C[i]C[i]C[i]的前缀和。

发现这个算法的瓶颈在寻找最优的jjj，如果我们可以O(1)O(1)O(1)的找到jjj，那么整个算法的时间复杂度为O(n)O(n)O(n)，可以AC。

代码实现（还是很简短的）：

注意此代码精度有问题，你们可以把除法改成乘法试一下。

#include <bits/stdc++.h>
#define MAXN 1000005
#define int long long
using namespace std;
inline int read(){
    int x=0,f=1;
    char ch=getchar();
    while (ch<'0'||ch>'9'){
        if (ch=='-') f=-1ll;
        ch=getchar();
    }
    while (ch>='0'&&ch<='9'){
        x=(x<<3ll)+(x<<1ll)+(ch^'0');
        ch=getchar();
    }
    return x*f;
}
int sum[MAXN];
int dp[MAXN],q[MAXN],head,rear;
inline double Slope(int i,int j){
    return (double)(dp[i]+sum[i]*sum[i]-dp[j]-sum[j]*sum[j])/(double)(sum[i]-sum[j]);
}
#undef int
int main(){
#define int long long
    int n,M;
    while (~scanf("%lld%lld",&n,&M)){
        memset(sum,0,sizeof(sum));
        for (register int i=1;i<=n;++i){
            int x;
            scanf("%lld",&x);
            sum[i]=sum[i-1]+x;
        }
        memset(dp,0,sizeof(dp));
        memset(q,0,sizeof(q));
        head=1,rear=1;
        q[1]=0;
        for (register int i=1;i<=n;++i){
            while (head<rear&&Slope(q[head+1],q[head])<=(double)2.00*sum[i]) head++;
            int j=q[head];
            dp[i]=dp[j]+M+(sum[i]-sum[j])*(sum[i]-sum[j]);
            while (head<rear&&Slope(q[rear],q[rear-1])>=Slope(i,q[rear])) rear--;
            q[++rear]=i;
        }
        printf("%lld\n",dp[n]);
    }
}
